1. SQL Supabase 
drop table if exists mise_en_place;
create table mise_en_place (
  id uuid default gen_random_uuid() primary key,
  section text not null,
  sous_section text,
  nom text not null,
  portions int not null default 0,
  rupture boolean not null default false,
  actif boolean not null default true,
  ordre int not null default 0,
  created_at timestamptz default now()
);
alter table mise_en_place enable row level security;
create policy "public_all" on mise_en_place
  for all to anon using (true) with check (true);
